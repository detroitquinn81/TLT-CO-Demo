    btn.classList.add('active');


 


    const cards = document.querySelectorAll('.menu-card');


    cards.forEach(card => {

        card.style.display = '';


      } else {


        card.classList.add('panel-hidden');


        card.style.display = 'none';


      }


    });


 


    // init tacos visible


    document.querySelectorAll('[data-cat="tacos"]').forEach(c => {


      if (cat === 'tacos') c.style.display = '';


    });


  }


 


  // On load show tacos


  document.querySelectorAll('[data-cat="tacos"]').forEach(c => c.style.display = '');


 


  // Reveal on scroll


  const observer = new IntersectionObserver((entries) => {


    entries.forEach(e => {


      if (e.isIntersecting) {


        e.target.classList.add('visible');


        observer.unobserve(e.target);


      }


    });


  }, { threshold: 0.12, rootMargin: '0px 0px -40px 0px' });


 


  document.querySelectorAll('.reveal').forEach(el => observer.observe(el));


 


  // Add btn interaction

        this.style.background = '';

    });


  });


 


  // Highlight today in hours (Fri/Sat)


  const day = new Date().getDay();


  const rows = document.querySelectorAll('.hours-table tr');


  rows.forEach(r => r.classList.remove('today'));


  if (day >= 1 && day <= 4) rows[0]?.classList.add('today');


  else if (day === 5 || day === 6) rows[1]?.classList.add('today');
